// Google Apps Script for All Love Mobile Muscle Tracker
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheetId = data.sheetId;
    const action = data.action;
    
    if (!sheetId) {
      return createResponse(400, { error: 'No sheet ID provided' });
    }
    
    const ss = SpreadsheetApp.openById(sheetId);
    
    if (action === 'setup') {
      // Create sheets if they don't exist
      let clientsSheet = ss.getSheetByName('Clients');
      if (!clientsSheet) {
        clientsSheet = ss.insertSheet('Clients');
        clientsSheet.getRange(1, 1, 1, 13).setValues([[
          'ID', 'First Name', 'Last Name', 'Phone', 'Email', 'Address', 
          'City', 'Zip Code', 'Type', 'Preferred Massage', 'Notes', 'Created At', 'Updated At'
        ]]);
      }
      
      let visitsSheet = ss.getSheetByName('Visits');
      if (!visitsSheet) {
        visitsSheet = ss.insertSheet('Visits');
        visitsSheet.getRange(1, 1, 1, 11).setValues([[
          'ID', 'Client ID', 'Date', 'Time', 'Type', 'Duration', 
          'Amount', 'Payment', 'Notes', 'Is Mobile', 'Created At'
        ]]);
      }
      
      let appointmentsSheet = ss.getSheetByName('Appointments');
      if (!appointmentsSheet) {
        appointmentsSheet = ss.insertSheet('Appointments');
        appointmentsSheet.getRange(1, 1, 1, 13).setValues([[
          'ID', 'Client ID', 'Date', 'Start Time', 'Duration', 'Type', 
          'Notes', 'Is Mobile', 'Status', 'Recurring', 'Created At', 'Updated At', 'Reminder Sent'
        ]]);
      }
      
      let wellnessSheet = ss.getSheetByName('Wellness');
      if (!wellnessSheet) {
        wellnessSheet = ss.insertSheet('Wellness');
        wellnessSheet.getRange(1, 1, 1, 12).setValues([[
          'ID', 'Date', 'Hours Worked', 'Sessions Completed', 'Breaks Taken', 
          'Water Intake', 'Pain Levels', 'Self Care Activities', 'Notes', 
          'Created At', 'Updated At', 'Wellness Score'
        ]]);
      }
      
      let selfcareSheet = ss.getSheetByName('SelfCare');
      if (!selfcareSheet) {
        selfcareSheet = ss.insertSheet('SelfCare');
        selfcareSheet.getRange(1, 1, 1, 14).setValues([[
          'ID', 'Date', 'Time', 'Duration', 'Type', 'Provider', 'Areas Treated',
          'Custom Areas', 'Techniques', 'Notes', 'Pain Before', 'Pain After',
          'Created At', 'Updated At'
        ]]);
      }
      
      return createResponse(200, { 
        success: true, 
        message: 'Sheets created successfully' 
      });
    }
    
    if (action === 'sync') {
      const clients = data.clients || [];
      const visits = data.visits || [];
      const appointments = data.appointments || [];
      const wellness = data.wellness || [];
      const selfcare = data.selfcare || [];
      
      // Update Clients sheet
      let clientsSheet = ss.getSheetByName('Clients');
      if (!clientsSheet) {
        clientsSheet = ss.insertSheet('Clients');
        clientsSheet.getRange(1, 1, 1, 13).setValues([[
          'ID', 'First Name', 'Last Name', 'Phone', 'Email', 'Address', 
          'City', 'Zip Code', 'Type', 'Preferred Massage', 'Notes', 'Created At', 'Updated At'
        ]]);
      }
      
      // Clear and update clients
      clientsSheet.getRange(2, 1, clientsSheet.getLastRow(), 13).clear();
      const clientData = clients.map(client => [
        client.id,
        client.firstName,
        client.lastName,
        client.phone,
        client.email || '',
        client.address || '',
        client.city || '',
        client.zipCode || '',
        client.type,
        client.preferredMassage,
        client.notes || '',
        client.createdAt,
        client.updatedAt || client.createdAt
      ]);
      
      if (clientData.length > 0) {
        clientsSheet.getRange(2, 1, clientData.length, 13).setValues(clientData);
      }
      
      // Update Visits sheet
      let visitsSheet = ss.getSheetByName('Visits');
      if (!visitsSheet) {
        visitsSheet = ss.insertSheet('Visits');
        visitsSheet.getRange(1, 1, 1, 11).setValues([[
          'ID', 'Client ID', 'Date', 'Time', 'Type', 'Duration', 
          'Amount', 'Payment', 'Notes', 'Is Mobile', 'Created At'
        ]]);
      }
      
      // Clear and update visits
      visitsSheet.getRange(2, 1, visitsSheet.getLastRow(), 11).clear();
      const visitData = visits.map(visit => [
        visit.id,
        visit.clientId,
        visit.date,
        visit.time,
        visit.type,
        visit.duration,
        visit.amount,
        visit.payment,
        visit.notes || '',
        visit.isMobile ? 'Yes' : 'No',
        visit.createdAt
      ]);
      
      if (visitData.length > 0) {
        visitsSheet.getRange(2, 1, visitData.length, 11).setValues(visitData);
      }
      
      // Update Appointments sheet
      let appointmentsSheet = ss.getSheetByName('Appointments');
      if (!appointmentsSheet) {
        appointmentsSheet = ss.insertSheet('Appointments');
        appointmentsSheet.getRange(1, 1, 1, 13).setValues([[
          'ID', 'Client ID', 'Date', 'Start Time', 'Duration', 'Type', 
          'Notes', 'Is Mobile', 'Status', 'Recurring', 'Created At', 'Updated At', 'Reminder Sent'
        ]]);
      }
      
      // Clear and update appointments
      appointmentsSheet.getRange(2, 1, appointmentsSheet.getLastRow(), 13).clear();
      const appointmentData = appointments.map(appt => [
        appt.id,
        appt.clientId,
        appt.date,
        appt.startTime,
        appt.duration,
        appt.type,
        appt.notes || '',
        appt.isMobile ? 'Yes' : 'No',
        appt.status,
        appt.recurring ? JSON.stringify(appt.recurring) : '',
        appt.createdAt,
        appt.updatedAt || appt.createdAt,
        appt.reminderSent || 'No'
      ]);
      
      if (appointmentData.length > 0) {
        appointmentsSheet.getRange(2, 1, appointmentData.length, 13).setValues(appointmentData);
      }
      
      // Update Wellness sheet
      let wellnessSheet = ss.getSheetByName('Wellness');
      if (!wellnessSheet) {
        wellnessSheet = ss.insertSheet('Wellness');
        wellnessSheet.getRange(1, 1, 1, 12).setValues([[
          'ID', 'Date', 'Hours Worked', 'Sessions Completed', 'Breaks Taken', 
          'Water Intake', 'Pain Levels', 'Self Care Activities', 'Notes', 
          'Created At', 'Updated At', 'Wellness Score'
        ]]);
      }
      
      // Clear and update wellness
      wellnessSheet.getRange(2, 1, wellnessSheet.getLastRow(), 12).clear();
      const wellnessData = wellness.map(well => [
        well.id,
        well.date,
        well.hoursWorked || 0,
        well.sessionsCompleted || 0,
        well.breaksTaken || 0,
        well.waterIntake || 0,
        JSON.stringify(well.painLevels || {}),
        well.selfCareActivities || '',
        well.notes || '',
        well.createdAt,
        well.updatedAt || well.createdAt,
        well.wellnessScore || 0
      ]);
      
      if (wellnessData.length > 0) {
        wellnessSheet.getRange(2, 1, wellnessData.length, 12).setValues(wellnessData);
      }
      
      // Update SelfCare sheet
      let selfcareSheet = ss.getSheetByName('SelfCare');
      if (!selfcareSheet) {
        selfcareSheet = ss.insertSheet('SelfCare');
        selfcareSheet.getRange(1, 1, 1, 14).setValues([[
          'ID', 'Date', 'Time', 'Duration', 'Type', 'Provider', 'Areas Treated',
          'Custom Areas', 'Techniques', 'Notes', 'Pain Before', 'Pain After',
          'Created At', 'Updated At'
        ]]);
      }
      
      // Clear and update selfcare
      selfcareSheet.getRange(2, 1, selfcareSheet.getLastRow(), 14).clear();
      const selfcareData = selfcare.map(sc => [
        sc.id,
        sc.date,
        sc.time,
        sc.duration,
        sc.type,
        sc.provider,
        JSON.stringify(sc.areas || []),
        sc.customAreas || '',
        sc.techniques || '',
        sc.notes || '',
        sc.painBefore || 0,
        sc.painAfter || 0,
        sc.createdAt,
        sc.updatedAt || sc.createdAt
      ]);
      
      if (selfcareData.length > 0) {
        selfcareSheet.getRange(2, 1, selfcareData.length, 14).setValues(selfcareData);
      }
      
      return createResponse(200, { 
        success: true, 
        message: 'Data synced successfully',
        syncedAt: new Date().toISOString()
      });
    }
    
    if (action === 'load') {
      const clientsSheet = ss.getSheetByName('Clients');
      const visitsSheet = ss.getSheetByName('Visits');
      const appointmentsSheet = ss.getSheetByName('Appointments');
      const wellnessSheet = ss.getSheetByName('Wellness');
      const selfcareSheet = ss.getSheetByName('SelfCare');
      
      let clients = [];
      let visits = [];
      let appointments = [];
      let wellness = [];
      let selfcare = [];
      
      if (clientsSheet && clientsSheet.getLastRow() > 1) {
        const clientRows = clientsSheet.getRange(2, 1, clientsSheet.getLastRow()-1, 13).getValues();
        clients = clientRows.map(row => ({
          id: row[0],
          firstName: row[1],
          lastName: row[2],
          phone: row[3],
          email: row[4] || null,
          address: row[5] || null,
          city: row[6] || null,
          zipCode: row[7] || null,
          type: row[8],
          preferredMassage: row[9],
          notes: row[10] || null,
          createdAt: row[11],
          updatedAt: row[12]
        }));
      }
      
      if (visitsSheet && visitsSheet.getLastRow() > 1) {
        const visitRows = visitsSheet.getRange(2, 1, visitsSheet.getLastRow()-1, 11).getValues();
        visits = visitRows.map(row => ({
          id: row[0],
          clientId: row[1],
          date: row[2],
          time: row[3],
          type: row[4],
          duration: row[5],
          amount: row[6],
          payment: row[7],
          notes: row[8] || null,
          isMobile: row[9] === 'Yes',
          createdAt: row[10]
        }));
      }
      
      if (appointmentsSheet && appointmentsSheet.getLastRow() > 1) {
        const appointmentRows = appointmentsSheet.getRange(2, 1, appointmentsSheet.getLastRow()-1, 13).getValues();
        appointments = appointmentRows.map(row => ({
          id: row[0],
          clientId: row[1],
          date: row[2],
          startTime: row[3],
          duration: row[4],
          type: row[5],
          notes: row[6] || null,
          isMobile: row[7] === 'Yes',
          status: row[8],
          recurring: row[9] ? JSON.parse(row[9]) : null,
          createdAt: row[10],
          updatedAt: row[11],
          reminderSent: row[12] === 'Yes'
        }));
      }
      
      if (wellnessSheet && wellnessSheet.getLastRow() > 1) {
        const wellnessRows = wellnessSheet.getRange(2, 1, wellnessSheet.getLastRow()-1, 12).getValues();
        wellness = wellnessRows.map(row => ({
          id: row[0],
          date: row[1],
          hoursWorked: row[2] || 0,
          sessionsCompleted: row[3] || 0,
          breaksTaken: row[4] || 0,
          waterIntake: row[5] || 0,
          painLevels: row[6] ? JSON.parse(row[6]) : {},
          selfCareActivities: row[7] || '',
          notes: row[8] || '',
          createdAt: row[9],
          updatedAt: row[10],
          wellnessScore: row[11] || 0
        }));
      }
      
      if (selfcareSheet && selfcareSheet.getLastRow() > 1) {
        const selfcareRows = selfcareSheet.getRange(2, 1, selfcareSheet.getLastRow()-1, 14).getValues();
        selfcare = selfcareRows.map(row => ({
          id: row[0],
          date: row[1],
          time: row[2],
          duration: row[3],
          type: row[4],
          provider: row[5],
          areas: row[6] ? JSON.parse(row[6]) : [],
          customAreas: row[7] || '',
          techniques: row[8] || '',
          notes: row[9] || '',
          painBefore: row[10] || 0,
          painAfter: row[11] || 0,
          createdAt: row[12],
          updatedAt: row[13]
        }));
      }
      
      return createResponse(200, { 
        success: true, 
        clients: clients,
        visits: visits,
        appointments: appointments,
        wellness: wellness,
        selfcare: selfcare,
        loadedAt: new Date().toISOString()
      });
    }
    
    return createResponse(400, { error: 'Invalid action' });
    
  } catch (error) {
    return createResponse(500, { error: error.toString() });
  }
}

function createResponse(code, data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
